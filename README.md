## 安装

```shell
npm i uw.ui -S
```

## 使用

```javascript
import Vue from "vue";
import UU from "uw.ui";

Vue.use(UU);

// 或者
import {
	Select,
	Button
	// ...
} from "uw.ui";

Vue.component(Select.name, Select);
Vue.component(Button.name, Button);
```
